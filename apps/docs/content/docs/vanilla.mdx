## Installation
<Tabs items={['pnpm', 'npm', 'yarn']} groupId="language">
  <Tab value="pnpm">
    ```bash
    pnpm add pinch-zoom-native
    ```  
  </Tab>
  <Tab value="npm">
    ```bash
    npm i pinch-zoom-native
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn add pinch-zoom-native
    ```
  </Tab>
</Tabs>

## Usage
```js
import { pinchZoom } from 'pinch-zoom-native'

const zoom = pinchZoom(element, options)
```
